        - align="" // Alineación del contenido de la celda
        - valign="" // Alineación vertical del contenido de la celda

    border="" // Bordes de la tabla
    cellpadding="" // Espacio entre el borde de la celda y su contenido
    cellspacing="" // Espacio entre celdas
    align="" // Alineación de la tabla
    width="" // Ancho de la tabla
    height="" // Altura de la tabla
    bgcolor="" // Color de fondo de la tabla
