```bash
 <ol> // Listas numéricas
    type="1" -> números (1, 2, 3, ...)
    type="A" -> letras mayúsculas (A, B, C, ...)
    type="a" -> letras minúsculas (a, b, c, ...)
    type="I" -> números romanos mayúsculas (I, II, III, ...)
    type="i" -> números romanos minúsculas (i, ii, iii, ...)
    start="n" -> para iniciar la numeración en número n
<ul> // Listas con viñetas
    type="disc" -> viñetas rellenas (predeterminado)
    type="circle" -> viñetas circulares
    type="square" -> viñetas cuadradas
<dl> // Listas de definición
    <dt> // término de definición
    <dd> // definición del término
```
