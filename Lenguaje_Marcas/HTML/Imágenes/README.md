```bash

<img> insertar imagenes en una pagina web
 // Atributos principales
    src = ruta de la imagen
    alt = texto alternativo (accesibilidad)
    width = ancho de la imagen
    height = alto de la imagen
    border = borde de la imagen (no recomendado, usar CSS)
    title = texto que aparece al pasar el cursor sobre la imagen
    align = alineacion de la imagen (no recomendado, usar CSS)
        - top = alinea la imagen con la parte superior del texto
        - middle = alinea la imagen con la parte media del texto
        - bottom = alinea la imagen con la parte inferior del texto
        - left = alinea la imagen a la izquierda del texto
        - right = alinea la imagen a la derecha del texto
```
